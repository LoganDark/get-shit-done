<objective>
Build the rewriter library that powers `/gsd-migrate-vcs`. This plan delivers a self-contained `sdk/src/vcs/format-migration/` package with zero CLI surface. Plan 06-03 wires this library into the SDK verb + workflow markdown. Plan 06-04 consumes the integrated stack for BROWN-01 dogfood.

The library implements the migration algorithm specified in RESEARCH §Pattern 1 + Pattern 2 + Pattern 3:

1. **Walk** the in-scope `.planning/` glob set via stdlib `fs.readdirSync({ recursive: true })` (RESEARCH §"Don't Hand-Roll" — no glob deps). Symlink guard via `fs.lstatSync` (RESEARCH §Security V4).
2. **Rewrite** each file's content via two pure regexes: `/(?<![0-9a-fA-F])([0-9a-f]{7,40})(?![0-9a-fA-F])/g` (git SHA) and `/(?<![k-z])([k-z]{8,12})(?![k-z])/g` (jj change_id). Direction selects which regex runs. ID alphabet disjointness confirmed by plan 06-01 A1 probe.
3. **Resolve** each match via cached `commitIdOf` / `changeIdOf` (existing API at `sdk/src/vcs/parse/jj-id.ts:33-67`). On `VcsExecError` → delegate to `orphan.resolveAncestor` which walks source-VCS ancestry via adapter calls until hitting a resolvable counterpart, then captures direct-children-in-target via `expr.children` (plan 06-01 Task 2 deliverable; jj-side only, git-side `children:` throws per plan 06-01 Task 2 deliverable).
4. **Emit report** to `.planning/intel/06-migration-report.md` (CONTEXT §"Specifics" sample row shape).
5. **Commit atomically** — single `vcs.commit({ files: [...dirty, config.json, report.md], message: '... [gsd-migrate-vcs v1]' })` per RESEARCH §Pattern 3.

Pre-flight refuses if dirty/conflicts and not `--force` (Pitfall 4). Post-flip adapter reconstructed with explicit `{ kind: target }` (Pitfall 6). Lock held for entire migration duration (Anti-pattern "Skipping the lock").

```typescript
import { commitIdOf, changeIdOf } from '../parse/jj-id.js';
import { expr } from '../expr.js';
import { VcsExecError } from '../exec.js';
import type { VcsAdapter } from '../types.js';
import type { MigrationDirection } from './types.js';

const MAX_DEPTH = 1000;  // safety bound

export async function resolveAncestor(
  vcs: VcsAdapter,
  cwd: string,
  orphan: string,
  direction: MigrationDirection,
): Promise<{ ancestor: string; childrenInTarget: string[] } | null> {
  // Plan 06-01 Task 2 ships `expr.parents(rev)` which translates to '(<inner>)-'
  // on jj and '<inner>^@' on git — both adapters can step one parent at a time
  // via vcs.log({ rev: expr.parents(expr.rev(cursor)), maxCount: 1 }). A5+A6
  // probes (plan 06-01 Task 3) confirm depth-1 semantics on jj 0.41.
  //
  // IMPORTANT: jj `jj log -r "<rev>"` returns only commits IN the revset, so the
  // earlier "vcs.log({rev: expr.rev(cursor), maxCount:2}) and take entries[1]"
  // trick only worked on git. With expr.parents we get symmetric behavior on
  // both backends.
  let cursor = orphan;
  for (let depth = 0; depth < MAX_DEPTH; depth++) {
    const parents = vcs.log({ rev: expr.parents(expr.rev(cursor)), maxCount: 1 });
    if (parents.length === 0) return null;  // hit root — no parent in source VCS
    cursor = parents[0].hash;

    // Probe whether cursor resolves in the TARGET VCS.
    try {
      const targetId = direction === 'git→jj'
        ? changeIdOf(cwd, cursor)
        : commitIdOf(cwd, cursor);
      // Direct children in TARGET VCS DAG. RESEARCH A5 (probed by plan 06-01)
      // confirms x+ is depth-1-only on jj. On the git direction (jj→git target
      // = git), expr.children throws (plan 06-01 Task 2) — so we leave the
      // children column empty on git target. Plan 06-01 Task 2's expr.parents
      // works on both backends, but expr.children is jj-only by design.
      let children: string[] = [];
      if (direction === 'git→jj') {
        const childEntries = vcs.log({ rev: expr.children(expr.rev(targetId)), maxCount: 100 });
        children = childEntries.map((c) => c.hash);
      } else {
        // git target — no children:'rev' translator on git side; emit empty list.
        // The migration report still records the ancestor; children column is left
        // empty for jj→git direction (documented in report.ts).
        children = [];
      }
      return { ancestor: targetId, childrenInTarget: children };
    } catch (e) {
      if (e instanceof VcsExecError && e.exitCode !== 0) continue;  // not in target — keep walking
      throw e;
    }
  }
  return null;
}
```

Per RESEARCH Pitfall 6, this is OK during the pre-flip phase (adapter still on source backend). The migration commit is the only operation that needs the post-flip adapter — orphan resolution happens BEFORE the flip.

**`run.ts`** — top-level orchestrator. Implements the 5-phase pipeline from RESEARCH §Pattern 3:

```typescript
import { readFile, writeFile } from 'node:fs/promises';
import { join } from 'node:path';
import { createVcsAdapter } from '../index.js';
import { atomicWriteConfig } from '../../query/config-mutation.js';
import { acquireStateLock, releaseStateLock } from '../../query/state-mutation.js';
import { planningPaths } from '../../query/helpers.js';
import { sanitizeCommitMessage } from '../../query/commit.js';
import { walkInScope } from './walk.js';
import { migrateContent } from './rewrite.js';
import { createIdResolver, syncResolveFromCache } from './resolve.js';
import { resolveAncestor } from './orphan.js';
import { emitReport } from './report.js';
import { MIGRATION_COMMIT_MARKER } from './types.js';
import type {
  MigrationDirection, MigrationResult, Orphan,
  ResolveResult, RunMigrationOpts,
} from './types.js';

export async function runMigration(
  cwd: string,
  target: 'git' | 'jj',
  opts: RunMigrationOpts = {},
): Promise<MigrationResult> {
  const paths = planningPaths(cwd, opts.workstream);
  const lockPath = await acquireStateLock(paths.config);
  try {
    // Pre-flight: determine current adapter + direction.
    const config = JSON.parse(await readFile(paths.config, 'utf-8').catch(() => '{}'));
    const previousAdapter = config?.vcs?.adapter ?? 'absent';
    const sourceKind: 'git' | 'jj' = previousAdapter === 'jj' ? 'jj' : 'git';
    if (sourceKind === target) {
      // Marker-probe fast-exit (RESEARCH Open Q #4 — Option A). If HEAD commit
      // already carries the [gsd-migrate-vcs v1] marker AND target matches
      // current adapter, no work to do. Cheaper than scanning all files for
      // source-VCS shapes and proves the prior migration ran to commit.
      try {
        const vcsProbe = createVcsAdapter(cwd, { kind: sourceKind });
        const head = await vcsProbe.log({ maxCount: 1 });
        const subject = head[0]?.message ?? head[0]?.subject ?? '';
        if (subject.includes(MIGRATION_COMMIT_MARKER)) {
          return {
            ok: true, migrated: false,
            filesChanged: 0, filesScanned: 0,
            orphans: { count: 0, ancestorResolved: 0, unresolvable: 0, reportPath: '' },
            previousAdapter: previousAdapter === 'absent' ? 'absent' : sourceKind,
            newAdapter: target,
            commitHash: head[0]?.hash ?? '',
          };
        }
      } catch { /* fall through to the explicit-already-on-target error */ }
      throw new Error(`migrate-vcs: already on ${target} (previousAdapter=${previousAdapter})`);
    }
    const direction: MigrationDirection = (sourceKind === 'git' && target === 'jj') ? 'git→jj' : 'jj→git';

    // Pre-flight: refuse on dirty / conflicts unless force (Pitfall 4).
    const vcs = createVcsAdapter(cwd, { kind: sourceKind });
    if (!opts.force) {
      const status = await vcs.status({ scope: 'working-copy' });
      if (status.entries.length > 0) {
        throw new Error('migrate-vcs: working tree is dirty — commit/stash or pass --force');
      }
      const conflicts = await vcs.findConflicts({ scope: 'all' });
      if (conflicts.length > 0) {
        throw new Error('migrate-vcs: in-tree conflicts present — resolve or pass --force');
      }
    }

    // Phase 1: walk + read + transform (in-memory).
    const files = walkInScope(cwd);
    // Async pre-pass: collect all matched IDs across all files, resolve them
    // via the cached async resolver, then call migrateContent synchronously
    // with the populated cache.
    const idCache = new Map<string, ResolveResult>();
    const asyncResolver = createIdResolver({
      cwd, vcs, direction,
      ancestor: resolveAncestor,
    });

    // Pre-pass: scan each file with the SAME regex used by migrateContent to
    // collect ID set, then resolve each via async API into the cache.
    const re = direction === 'git→jj'
      ? /(?<![0-9a-fA-F])([0-9a-f]{7,40})(?![0-9a-fA-F])/g
      : /(?<![k-z])([k-z]{8,12})(?![k-z])/g;
    const fileContents = new Map<string, string>();
    for (const f of files) {
      const content = await readFile(f, 'utf-8');
      fileContents.set(f, content);
      re.lastIndex = 0;
      let m: RegExpExecArray | null;
      while ((m = re.exec(content))) {
        if (!idCache.has(m[1])) idCache.set(m[1], await asyncResolver.resolve(m[1]));
      }
    }

    // Now run the SYNC transform over each file.
    const syncResolve = syncResolveFromCache(idCache);
    const dirty = new Map<string, { content: string; orphans: Orphan[] }>();
    let totalOrphans: Orphan[] = [];
    for (const f of files) {
      const original = fileContents.get(f)!;
      const out = migrateContent(original, direction, syncResolve, f);
      if (out.content !== original) {
        dirty.set(f, out);
      }
      totalOrphans.push(...out.orphans);
    }

    // Phase 2: write all dirty files.
    for (const [f, { content }] of dirty) {
      await writeFile(f, content, 'utf-8');
    }

    // Phase 3: flip config.json's vcs.adapter atomically.
    const newConfig = { ...config, vcs: { ...(config.vcs ?? {}), adapter: target } };
    await atomicWriteConfig(paths.config, newConfig);

    // Phase 4: emit report.
    const reportPath = await emitReport({
      cwd, direction,
      orphans: totalOrphans,
      filesScanned: files.length,
      filesChanged: dirty.size,
    });

    // Phase 5: single atomic commit on the TARGET backend (Pitfall 6).
    const newVcs = createVcsAdapter(cwd, { kind: target });
    // Workaround for A3 colocated pre-commit hook gap on jj (RESEARCH Open Q #5).
    if (target === 'jj') {
      await newVcs.hooks.fire('pre-commit', { cwd, stage: 'pre-commit' });
    }
    const message = sanitizeCommitMessage(
      `chore(vcs): migrate ${sourceKind} -> ${target} ${MIGRATION_COMMIT_MARKER}`
    );
    const commitResult = await newVcs.commit({
      files: [...dirty.keys(), paths.config, reportPath],
      message,
    });

    const ancestorCount = totalOrphans.filter((o) => o.kind === 'ancestor').length;
    const unresolvable = totalOrphans.filter((o) => o.kind === 'unresolvable').length;
    return {
      ok: true,
      migrated: true,
      filesChanged: dirty.size,
      filesScanned: files.length,
      orphans: {
        count: totalOrphans.length,
        ancestorResolved: ancestorCount,
        unresolvable,
        reportPath,
      },
      previousAdapter: previousAdapter === 'absent' ? 'absent' : sourceKind,
      newAdapter: target,
      commitHash: commitResult.hash ?? '',
    };
  } finally {
    await releaseStateLock(lockPath);
  }
}
```
