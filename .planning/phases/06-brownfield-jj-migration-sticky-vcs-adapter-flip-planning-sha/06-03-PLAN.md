**File 1 — `sdk/src/query/migrate-vcs.ts` (NEW):**

SDK verb handler. Argv parsing + current-state-aware defaults + pre-flight + dispatch to `runMigration` from plan 06-02:

```typescript
import { readFile } from 'node:fs/promises';
import { execSync } from 'node:child_process';
import { join } from 'node:path';
import { runMigration } from '../vcs/format-migration/index.js';
import { planningPaths } from './helpers.js';
import type { QueryHandler } from './utils.js';

const VALID_TARGETS = new Set(['git', 'jj']);

export const migrateVcsQuery: QueryHandler = async (args, projectDir) => {
  let cwd = projectDir;
  let target: string | undefined;
  let native = false;
  let force = false;
  let workstream: string | undefined;

  for (let i = 0; i < args.length; i++) {
    if (args[i] === '--cwd' && args[i + 1]) {
      cwd = args[i + 1];
      i++;
    } else if (args[i] === '--target' && args[i + 1]) {
      target = args[i + 1];
      i++;
    } else if (args[i] === '--workstream' && args[i + 1]) {
      workstream = args[i + 1];
      i++;
    } else if (args[i] === '--native') {
      native = true;
    } else if (args[i] === '--force') {
      force = true;
    } else if (args[i].startsWith('--')) {
      return { data: { ok: false, error: `migrate-vcs: unknown flag '${args[i]}'` } };
    }
  }

  // Determine current adapter (RESEARCH §"Determining the target VCS" table).
  const paths = planningPaths(cwd, workstream);
  let currentAdapter: 'git' | 'jj' | 'auto' | 'absent' = 'absent';
  try {
    const raw = await readFile(paths.config, 'utf-8');
    const json = JSON.parse(raw);
    currentAdapter = json?.vcs?.adapter ?? 'absent';
  } catch { /* leave 'absent' */ }

  // Current-state-aware target default (CONTEXT D-03).
  if (target === undefined) {
    if (currentAdapter === 'git' || currentAdapter === 'absent' || currentAdapter === 'auto') {
      target = 'jj';
    } else if (currentAdapter === 'jj') {
      return { data: { ok: false, error: 'migrate-vcs: already on jj — pass --target git to migrate back' } };
    }
  }

  // Validate target (RESEARCH §Security V5).
  if (!VALID_TARGETS.has(target!)) {
    return { data: { ok: false, error: `migrate-vcs: invalid --target '${target}' (valid: jj, git)` } };
  }

  // Refuse same-direction migration (currentAdapter === target).
  if (target === currentAdapter) {
    return { data: { ok: false, error: `migrate-vcs: already on ${target}` } };
  }

  // Pre-flight: target=jj requires jj binary available.
  if (target === 'jj') {
    try {
      execSync('jj --version', { stdio: 'pipe' });
    } catch {
      return { data: { ok: false, error: 'migrate-vcs: --target jj requires jj binary in PATH (install jj first)' } };
    }
  }

  // Dispatch to plan 06-02 library.
  try {
    const result = await runMigration(cwd, target as 'git' | 'jj', { force, native, workstream });
    // Flat envelope (CR-01 — no .data wrapper at the user-visible level beyond
    // the SDK's QueryHandler return shape). The QueryHandler return type IS
    // { data: ... } but the result fields are flat under that wrapper.
    return {
      data: {
        ok: true,
        migrated: result.migrated,
        filesChanged: result.filesChanged,
        filesScanned: result.filesScanned,
        orphans: result.orphans,
        previousAdapter: result.previousAdapter,
        newAdapter: result.newAdapter,
        commitHash: result.commitHash,
      },
    };
  } catch (e) {
    return { data: { ok: false, error: `migrate-vcs: ${(e as Error).message}` } };
  }
};
```

```typescript
// ─── initMigrateVcs ───────────────────────────────────────────────────────

/**
 * Init handler for migrate-vcs workflow (Phase 6 plan 06-03).
 * Returns pre-flight probe data: has_git, has_jj, current_adapter, jj_available,
 * dirty, conflicts, project_path, commit_docs.
 */
export const initMigrateVcs: QueryHandler = async (_args, projectDir) => {
  const config = await loadConfig(projectDir);
  let currentAdapter: 'git' | 'jj' | 'auto' | 'absent' = 'absent';
  try {
    const raw = await readFile(join(projectDir, '.planning', 'config.json'), 'utf-8');
    const json = JSON.parse(raw);
    currentAdapter = json?.vcs?.adapter ?? 'absent';
  } catch { /* leave 'absent' */ }

  let jjAvailable = false;
  try { execSync('jj --version', { stdio: 'pipe' }); jjAvailable = true; } catch { /* false */ }

  // Probe working-tree cleanliness via the adapter (avoids raw git; preserves no-raw-git invariant).
  let dirty = false;
  let conflicts = false;
  try {
    const vcs = createVcsAdapter(projectDir);
    const status = await vcs.status({ scope: 'working-copy' });
    dirty = (status?.entries?.length ?? 0) > 0;
    const conflictList = await vcs.findConflicts({ scope: 'all' });
    conflicts = (conflictList?.length ?? 0) > 0;
  } catch { /* if adapter construction fails, leave both false */ }

  const result: Record<string, unknown> = {
    has_git: pathExists(projectDir, '.git'),
    has_jj: pathExists(projectDir, '.jj'),
    current_adapter: currentAdapter,
    jj_available: jjAvailable,
    dirty,
    conflicts,
    project_path: projectDir,
    commit_docs: config.commit_docs,
  };
  return { data: withProjectRoot(projectDir, result, config as Record<string, unknown>) };
};
```
