```javascript
/**
 * Phase 4 plan 02: multi-workspace contract fixture.
 *
 * Like vcsTest, but inside the per-describe `before`, creates `n` workspaces
 * under .claude/jj-workspaces/phase-04-subagent-{idx} BEFORE running suiteFn.
 * suiteFn receives a handle exposing the workspace names via getWorkspaces().
 *
 * Mirrors the workspace-name convention from CONTEXT D-04 (zero-padded phase,
 * 1-indexed subagent).
 *
 * @param {'git'|'jj-colocated'|'jj-native'|Array<'git'|'jj-colocated'|'jj-native'>} kindOrKinds
 * @param {number} n - number of workspaces to pre-create
 * @param {(handle: { vcs: VcsAdapter, cwd: string, getWorkspaces: () => string[] }) => void} suiteFn
 */
function vcsMultiWsTest(kindOrKinds, n, suiteFn) {
  // Reuse vcsTest's mechanism: invoke vcsTest with a wrapping suiteFn that
  // pre-creates n workspaces before calling the user's suiteFn.
  vcsTest(kindOrKinds, (handle) => {
    const workspaceNames = [];
    const { join } = require('node:path');
    const { mkdirSync } = require('node:fs');
    // Hook into the existing fixture's `before`/`beforeEach`. The simplest
    // approach: declare a `before` here that uses handle.vcs (resolved at
    // describe time).
    before(() => {
      for (let i = 1; i <= n; i++) {
        const wsName = `phase-04-subagent-${i}`;
        const wsPath = join(handle.cwd, '.claude/jj-workspaces', wsName);
        // workspace.add does its own mkdir -p; we do not pre-create here.
        handle.vcs.workspace.add({ path: wsPath, name: wsName });
        workspaceNames.push(wsName);
      }
    });
    after(() => {
      // Best-effort cleanup; tests that explicitly forget should still pass.
      for (const wsName of workspaceNames) {
        try { handle.vcs.workspace.forget(wsName); } catch { /* already forgotten */ }
      }
    });
    handle.getWorkspaces = () => [...workspaceNames];
    suiteFn(handle);
  });
}

module.exports.vcsMultiWsTest = vcsMultiWsTest;
```
