```typescript
const log = (opts: LogOpts = {}): LogEntry[] => {
  const args: string[] = ['log', '-T', 'json(self) ++ "\\n"', '--no-graph'];
  if (opts.maxCount) args.push('-n', String(opts.maxCount));
  if (opts.allRefs) args.push('-r', 'all()');
  if (opts.rev) args.push('-r', toJjRev(opts.rev));
  if (opts.paths && opts.paths.length > 0) args.push(...opts.paths);
  const r = vcsExec(cwd, 'jj', jjArgv(...args));
  if (r.exitCode !== 0) return [];
  return parseJjLog(r.stdout);
};

const status = (opts: StatusOpts = {}): StatusResult => {
  const args = jjArgv('status');
  const r = vcsExec(cwd, 'jj', args);
  if (r.exitCode !== 0) return { entries: [], raw: r.stderr || r.stdout };
  if (opts.porcelain === false) {
    return { entries: [], raw: r.stdout };
  }
  return { entries: parseJjStatus(r.stdout), raw: r.stdout };
};

const diff = (opts: DiffOpts = {}): DiffResult => {
  const args: string[] = ['diff'];
  if (opts.nameOnly) args.push('--name-only');
  if (opts.nameStatus) args.push('--summary');
  if (opts.rev) args.push('-r', toJjRev(opts.rev));
  if (opts.paths && opts.paths.length > 0) args.push(...opts.paths);
  // opts.staged: no-op on jj (no index). Documented in JSDoc above this method.
  const r = vcsExec(cwd, 'jj', jjArgv(...args));
  if (r.exitCode !== 0) {
    return { raw: r.stderr || r.stdout, nameOnly: [] };
  }
  const result: DiffResult = {
    raw: r.stdout,
    nameOnly: opts.nameOnly
      ? r.stdout.split('\n').map((s) => s.trim()).filter(Boolean)
      : [],
  };
  if (opts.nameStatus) {
    result.nameStatus = parseDiffSummary(r.stdout);
  }
  return result;
};
```

```typescript
/**
 * Parse `jj status` human-readable output into StatusEntry[].
 *
 * Phase 2.1 D-16: StatusEntry has NO `index` field ŌĆö jj has no index. The
 * `worktree` letter is whatever jj prints (A/M/D/R/C).
 *
 * Output sample (jj 0.41):
 *   Working copy changes:
 *   A a.txt
 *   M b.txt
 *   Working copy  (@) : wttxkypv e4595a81 (no description set)
 *   Parent commit (@-): ...
 */
const parseJjStatus = (raw: string): StatusEntry[] => {
  const lines = raw.split('\n');
  const entries: StatusEntry[] = [];
  let inSection = false;
  for (const line of lines) {
    if (line.startsWith('Working copy changes:')) { inSection = true; continue; }
    if (line.startsWith('Working copy  (@)') || line.startsWith('Parent commit')) break;
    if (!inSection) continue;
    const m = /^([AMDRC]) (.+)$/.exec(line);
    if (m) entries.push({ path: m[2], worktree: m[1] });
  }
  return entries;
};

/**
 * Parse `jj diff --summary` line-by-line into DiffNameStatusEntry[].
 * Output sample: `M path/to/file.ts`, `A other.txt`, etc.
 */
const parseDiffSummary = (raw: string): DiffNameStatusEntry[] => {
  const entries: DiffNameStatusEntry[] = [];
  for (const line of raw.split('\n')) {
    const m = /^([AMDRCTUXB]) (.+)$/.exec(line);
    if (m) {
      entries.push({ path: m[2], status: m[1] as DiffNameStatusEntry['status'] });
    }
  }
  return entries;
};
```

```typescript
const findConflicts = (
  opts: { scope: 'all' | 'working-copy' }
): ConflictResult[] => {
  // ŌÜĀ’ĖÅ CRITICAL: jj's revset function is `conflicts()` PLURAL, not
  // `conflict()`. CONTEXT.md / REQUIREMENTS.md / ROADMAP.md all incorrectly
  // use the singular form; the doc-fix lands in plan 03-07 wrap-up.
  // Implementation here uses the correct plural form from day one.
  // See 03-RESEARCH.md ┬¦"Open Question Q1" for verification.
  const revset = opts.scope === 'working-copy'
    ? 'conflicts() & @'
    : 'conflicts()';
  const logArgs = jjArgv(
    'log',
    '-r', revset,
    '-T', 'json(self) ++ "\\n"',
    '--no-graph'
  );
  const r = vcsExec(cwd, 'jj', logArgs);
  if (r.exitCode !== 0) return [];
  const entries = parseJjLog(r.stdout);
  if (entries.length === 0) return [];

  const results: ConflictResult[] = [];
  for (const entry of entries) {
    // RESEARCH A3 (medium-risk): `jj resolve --list -r <rev>` is the
    // primary path-enumeration form. Fallback path: `jj diff -r <rev>
    // --summary` filtered for conflict-marker entries. Implementation
    // attempts primary first; on failure, falls back.
    const paths = enumerateConflictedPaths(entry.hash);
    results.push({ rev: entry.hash, paths, scope: opts.scope });
  }
  return results;
};

/**
 * Phase 3 RESEARCH A3 (medium-risk): enumerate conflicted paths for a
 * given revision. Primary form: `jj resolve --list -r <rev>`. Fallback:
 * `jj diff -r <rev> --summary` filtered for `C` (conflict) entries.
 * If both fail, returns []; the conflict-detection itself already succeeded
 * via the revset query ŌĆö paths-empty is a soft degradation, not a wrong-answer.
 */
const enumerateConflictedPaths = (rev: string): string[] => {
  // Primary: jj resolve --list -r <rev>
  const primaryArgs = jjArgv('resolve', '--list', '-r', rev);
  const primary = vcsExec(cwd, 'jj', primaryArgs);
  if (primary.exitCode === 0 && primary.stdout.trim().length > 0) {
    // `jj resolve --list` output sample: lines of `<path> <conflict-marker>`
    // or just `<path>` ŌĆö strip everything after the first space if jj
    // emits annotated form. Verify locally during plan execution.
    return primary.stdout
      .split('\n')
      .map((s) => s.trim())
      .filter(Boolean)
      .map((line) => {
        // If line has trailing diagnostic info, keep only the path portion.
        const m = /^(\S+)/.exec(line);
        return m ? m[1] : line;
      });
  }
  // Fallback: jj diff -r <rev> --summary, filter for status === 'C' or 'U'
  const fallbackArgs = jjArgv('diff', '-r', rev, '--summary');
  const fallback = vcsExec(cwd, 'jj', fallbackArgs);
  if (fallback.exitCode !== 0) return [];
  return fallback.stdout
    .split('\n')
    .filter(Boolean)
    .map((line) => {
      const m = /^[CU] (.+)$/.exec(line);
      return m ? m[1] : '';
    })
    .filter(Boolean);
};
```
