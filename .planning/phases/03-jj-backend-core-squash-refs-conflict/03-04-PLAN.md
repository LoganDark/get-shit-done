```typescript
const commit = (input: CommitInput): CommitResult => {
  if (input.files !== undefined && input.files.length === 0) {
    throw new Error('commit({files:[]}) is ambiguous; pass files: undefined for the all-changes form, or pass at least one path to commit a specific path set.');
  }
  if (input.amend) throw new VcsNotImplementedError('amend: not yet supported on jj backend (deferred per Phase 3 §Q5)');
  const squashArgs = jjArgv('squash', '-B', '@', '-k', '-m', input.message);
  if (input.files && input.files.length > 0) squashArgs.push(...input.files);
  const squashRes = vcsExec(cwd, 'jj', squashArgs, envOpts());
  if (squashRes.exitCode !== 0) {
    return { exitCode: squashRes.exitCode, stdout: squashRes.stdout, stderr: squashRes.stderr, hash: null };
  }
  const hashRes = vcsExec(cwd, 'jj', jjArgv('log', '-r', '@-', '-T', 'commit_id', '--no-graph', '-n', '1'));
  if (input.bookmark || input.bookmarkRaw) {
    const bmName = input.bookmarkRaw ?? addPrefix(input.bookmark!);
    const advRes = vcsExec(cwd, 'jj', jjArgv('bookmark', 'set', bmName, '-r', '@-', '-B'));
    if (advRes.exitCode !== 0) {
      return { ...squashRes, stderr: `${squashRes.stderr}\n${advRes.stderr}`, hash: hashRes.exitCode === 0 ? hashRes.stdout.trim() : null };
    }
  }
  return { exitCode: squashRes.exitCode, stdout: squashRes.stdout, stderr: squashRes.stderr, hash: hashRes.exitCode === 0 ? hashRes.stdout.trim() : null };
};
```

3. **commit body:**

```typescript
const commit = (input: CommitInput): CommitResult => {
  // WR-01 verbatim from git.ts:95-100 (same ambiguity is cross-backend):
  if (input.files !== undefined && input.files.length === 0) {
    throw new Error(
      'commit({files:[]}) is ambiguous; pass files: undefined for the all-changes form, ' +
        'or pass at least one path to commit a specific path set.',
    );
  }
  if (input.amend) {
    throw new VcsNotImplementedError(
      'amend: not yet supported on jj backend (deferred per Phase 3 RESEARCH §Q5)',
    );
  }
  // allowEmpty: jj squash naturally no-ops on empty source; document and ignore.
  // noVerify: Phase 4 owns hook firing; no-op here.

  // SQUASH-01 / SQUASH-02: argv-array invocation, files trailing positional.
  const squashArgs = jjArgv('squash', '-B', '@', '-k', '-m', input.message);
  if (input.files && input.files.length > 0) {
    squashArgs.push(...input.files);
  }
  const squashRes = vcsExec(cwd, 'jj', squashArgs, envOpts());
  if (squashRes.exitCode !== 0) {
    return {
      exitCode: squashRes.exitCode,
      stdout: squashRes.stdout,
      stderr: squashRes.stderr,
      hash: null,
    };
  }

  // Hash resolution — after `-B @ -k`, the new commit is at @-.
  // RESEARCH §commit(): use a second `jj log -r @- -T commit_id -n 1` call;
  // parsing `Created new commit ...` stdout is fragile.
  const hashArgs = jjArgv(
    'log', '-r', '@-', '-T', 'commit_id', '--no-graph', '-n', '1'
  );
  const hashRes = vcsExec(cwd, 'jj', hashArgs);
  const hash = hashRes.exitCode === 0 ? hashRes.stdout.trim() : null;

  // D-01 / D-04: bookmark advance.
  if (input.bookmark !== undefined || input.bookmarkRaw !== undefined) {
    const bmName = input.bookmarkRaw !== undefined
      ? input.bookmarkRaw
      : addPrefix(input.bookmark!);
    const advArgs = jjArgv('bookmark', 'set', bmName, '-r', '@-', '-B');
    const advRes = vcsExec(cwd, 'jj', advArgs);
    if (advRes.exitCode !== 0) {
      return {
        exitCode: squashRes.exitCode,
        stdout: squashRes.stdout,
        stderr: `${squashRes.stderr}\n[bookmark advance failed]: ${advRes.stderr || advRes.stdout}`,
        hash,
      };
    }
  }

  return {
    exitCode: squashRes.exitCode,
    stdout: squashRes.stdout,
    stderr: squashRes.stderr,
    hash,
  };
};
```
