  // ─── refs.bookmarks namespace ──────────────────────────────────────────
  const bookmarks: VcsBookmarks = Object.freeze({
    list: (): Bookmark[] => {
      const args = jjArgv('bookmark', 'list', '-T', 'json(self) ++ "\\n"');
      const r = vcsExec(cwd, 'jj', args);
      if (r.exitCode !== 0) {
        throw new VcsExecError({
          message: `refs.bookmarks.list failed: ${r.stderr || r.stdout}`,
          bin: 'jj', args, exitCode: r.exitCode, stdout: r.stdout, stderr: r.stderr, timedOut: r.timedOut,
        });
      }
      const lines = r.stdout.split('\n').filter(Boolean);
      const entries: Bookmark[] = [];
      for (const line of lines) {
        const record = JSON.parse(line);
        if (Array.isArray(record.target) && record.target.length > 1) {
          throw new VcsBookmarkDivergentError({
            bookmarkName: record.name,
            divergentTargets: record.target as readonly string[],
          });
        }
        entries.push({ name: stripPrefix(record.name), rev: record.target?.[0] ?? '' });
      }
      return entries;
    },
    create: (name: string, rev: RevisionExpr, opts?: { raw?: boolean }): void => {
      const actualName = addPrefix(name, opts?.raw);
      const args = jjArgv('bookmark', 'create', actualName, '-r', toJjRev(rev));
      const r = vcsExec(cwd, 'jj', args);
      if (r.exitCode !== 0) {
        throw new Error(`refs.bookmarks.create failed: ${r.stderr || r.stdout}`);
      }
    },
    move: (name: string, rev: RevisionExpr, opts?: { raw?: boolean }): void => {
      const actualName = addPrefix(name, opts?.raw);
      const args = jjArgv('bookmark', 'move', actualName, '--to', toJjRev(rev));
      const r = vcsExec(cwd, 'jj', args);
      if (r.exitCode !== 0) {
        throw new Error(`refs.bookmarks.move failed: ${r.stderr || r.stdout}`);
      }
    },
    delete: (name: string, opts?: { raw?: boolean }): void => {
      const actualName = addPrefix(name, opts?.raw);
      const args = jjArgv('bookmark', 'delete', actualName);
      const r = vcsExec(cwd, 'jj', args);
      if (r.exitCode !== 0) {
        throw new Error(`refs.bookmarks.delete failed: ${r.stderr || r.stdout}`);
      }
    },
    exists: (name: string, opts?: { raw?: boolean }): boolean => {
      const actualName = addPrefix(name, opts?.raw);
      const args = jjArgv('bookmark', 'list', actualName);
      const r = vcsExec(cwd, 'jj', args);
      return r.exitCode === 0 && r.stdout.trim().length > 0;
    },
    switch: (_name: string, _opts?: { create?: boolean; raw?: boolean }): void => {
      // RESEARCH §`refs.bookmarks.switch`: no Phase 3 caller exercises this on jj;

      throw new VcsNotImplementedError(

      );
    },
  });

  // ─── refs namespace ────────────────────────────────────────────────────
  const refs: VcsRefs = Object.freeze({
    head: expr.head(),
    parent: expr.parent(),
    bookmarks,

    currentBookmarks: (): string[] => {
      const args = jjArgv('log', '-r', '@-', '-T', 'bookmarks.join("\\n")', '--no-graph', '-n', '1');
      const r = vcsExec(cwd, 'jj', args);
      if (r.exitCode !== 0) return [];
      return r.stdout
        .split('\n')
        .map((s) => s.trim())
        .filter(Boolean)
        .map((s) => s.replace(/\*$/, '')) // strip optional local-divergent marker
        .map(stripPrefix);
    },

    resolveShort: (rev: RevisionExpr): string => {
      const args = jjArgv('log', '-r', toJjRev(rev), '-T', 'commit_id.short()', '--no-graph', '-n', '1');
      const r = vcsExec(cwd, 'jj', args);
      if (r.exitCode !== 0) {
        throw new Error(`refs.resolveShort failed: ${r.stderr || r.stdout}`);
      }
      return r.stdout.trim();
    },

    countCommits: ({ rev }: { rev?: RevisionExpr }): number => {
      const target = rev ? toJjRev(rev) : '::@';
      const args = jjArgv('log', '-r', target, '-T', '"\\n"', '--no-graph');
      const r = vcsExec(cwd, 'jj', args);
      if (r.exitCode !== 0) return 0;
      // Count newlines in the trimmed stdout (each commit emits exactly one "\n").
      const n = (r.stdout.match(/\n/g) ?? []).length;
      return Number.isFinite(n) ? n : 0;
    },

    rootCommits: ({ rev }: { rev?: RevisionExpr }): string[] => {
      const target = rev ? toJjRev(rev) : '@';
      const args = jjArgv(
        'log',
        '-r',
        `root() & ::${target}`,
        '-T',
        'commit_id ++ "\\n"',
        '--no-graph'
      );
      const r = vcsExec(cwd, 'jj', args);
      if (r.exitCode !== 0) return [];
      return r.stdout.split('\n').filter(Boolean);
    },

    exists: (rev: RevisionExpr): boolean => {
      const args = jjArgv('log', '-r', toJjRev(rev), '-T', '"x"', '--no-graph', '-n', '1');
      const r = vcsExec(cwd, 'jj', args);
      return r.exitCode === 0 && r.stdout.trim().length > 0;
    },

    isIgnored: (_path: string): boolean => {

      const r = vcsExec(cwd, 'jj', args);
      if (r.exitCode !== 0) return [];
      return r.stdout.split('\n').filter(Boolean);
    },
  });

// sdk/src/vcs/parse/jj-bookmark.ts (NEW)
import type { Bookmark } from '../types.js';
import { VcsBookmarkDivergentError } from '../types.js';

export function parseJjBookmarkRecord(line: string, stripPrefix: (s: string) => string): Bookmark {
  const record = JSON.parse(line);
  if (Array.isArray(record.target) && record.target.length > 1) {
    throw new VcsBookmarkDivergentError({
      bookmarkName: record.name,
      divergentTargets: record.target as readonly string[],
    });
  }
  return { name: stripPrefix(record.name), rev: record.target?.[0] ?? '' };
}
