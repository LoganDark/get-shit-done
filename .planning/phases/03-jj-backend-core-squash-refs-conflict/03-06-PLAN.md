const push = (opts: PushOpts = {}): ExecResult => {
  const args: string[] = ['git', 'push'];
  if (opts.remote) args.push('--remote', opts.remote);
  if (opts.ref) {
    // RESEARCH §push: jj pushes bookmarks, not arbitrary revs. If the ref is
    // a bookmark RevisionExpr, extract the name. For other ref shapes, log
    // a warning and proceed without --bookmark (jj's default push behavior).
    const refName = toJjRev(opts.ref);
    // jj's bookmark refs in toJjRev typically come through as the literal
    // bookmark name; if the form looks like a path/revset (`@`, `@-`, etc.),
    // skip the --bookmark arg.
    const isBookmarkLike = /^[A-Za-z][\w\-/.]*$/.test(refName);
    if (isBookmarkLike) {
      args.push('--bookmark', refName);
    } else {
      // No std-err writes here — return the warning in stderr via the
      // ExecResult so callers can surface it. Add a comment for the
      // future-Phase-4 review.
    }
  }
  if (opts.force) {
    args.push('--force-with-lease'); // RESEARCH A4 — verified during plan-06 empirical step
  }
  // opts.noVerify: Phase 4 owns hook firing; no-op here.
  return vcsExec(cwd, 'jj', jjArgv(...args));
};

const fetch = (opts: FetchOpts = {}): ExecResult => {
  const args: string[] = ['git', 'fetch'];

  }
  return vcsExec(cwd, 'jj', jjArgv(...args));
};
